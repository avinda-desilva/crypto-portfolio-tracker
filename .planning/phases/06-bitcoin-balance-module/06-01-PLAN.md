```typescript
// eslint-disable-next-line @typescript-eslint/no-require-imports
const { getBitcoinBalance } = require("../../lib/bitcoin");

describe("getBitcoinBalance", () => {
  let fetchSpy: jest.SpyInstance;

  beforeEach(() => {
    jest.clearAllMocks();
    // Restore any prior spy before establishing a fresh one each test
    fetchSpy = jest.spyOn(global, "fetch");
  });

  afterEach(() => {
    fetchSpy.mockRestore();
  });

  it("returns BTC balance as a number for a valid address", async () => {
    // (150_000_000 - 50_000_000) / 1e8 = 1.0 BTC
    fetchSpy.mockResolvedValue({
      ok: true,
      json: async () => ({
        chain_stats: {
          funded_txo_sum: 150000000,
          spent_txo_sum: 50000000,
        },
      }),
    } as unknown as Response);

    const result = await getBitcoinBalance("1A1zP1eP5QGefi2DMPTfTL5SLmv7Divf");
    expect(typeof result).toBe("number");
    expect(result).toBeCloseTo(1.0);
  });

  it("returns 0 when fetch throws a network error", async () => {
    fetchSpy.mockRejectedValue(new Error("network error"));
    const result = await getBitcoinBalance("1A1zP1eP5QGefi2DMPTfTL5SLmv7Divf");
    expect(result).toBe(0);
  });

  it("returns 0 on a non-OK HTTP response (e.g. 404)", async () => {
    fetchSpy.mockResolvedValue({
      ok: false,
      status: 404,
    } as unknown as Response);

    const result = await getBitcoinBalance("1A1zP1eP5QGefi2DMPTfTL5SLmv7Divf");

  });

  it("returns 0 when response body is missing chain_stats", async () => {
    fetchSpy.mockResolvedValue({
```

```typescript
/**
 * Bitcoin balance fetcher via the Blockstream public API.
 * No API key required. Uses native fetch (available in Next.js server environment).
 *
 * Mirrors the structure of lib/ethereum.ts:
 *   - Named export only (no default export)
 *   - Returns 0 on any error (never throws)
 *   - Logs failures with sanitized address context
 */

const BLOCKSTREAM_API = "https://blockstream.info/api/address";

interface BlockstreamResponse {
  chain_stats?: {
    funded_txo_sum?: number;
    spent_txo_sum?: number;
  };
}

// Named export only — no default export (consistent with lib/ethereum.ts and project conventions)
export async function getBitcoinBalance(address: string): Promise<number> {
  // Sanitize address to prevent log injection via crafted newline/ANSI sequences
  const safeAddress = address.replace(/[^\w.:-]/g, "?");
  try {
    const response = await fetch(`${BLOCKSTREAM_API}/${address}`);

    if (!response.ok) {
      console.error(
        `[bitcoin] getBitcoinBalance failed for address ${safeAddress}: HTTP ${response.status}`
      );
      return 0;
    }

    const data: BlockstreamResponse = await response.json() as BlockstreamResponse;

    const funded = data?.chain_stats?.funded_txo_sum;
    const spent = data?.chain_stats?.spent_txo_sum;

    if (typeof funded !== "number" || typeof spent !== "number") {
      console.error(
        `[bitcoin] getBitcoinBalance failed for address ${safeAddress}: unexpected response structure`,
        data
      );
      return 0;
    }

    // Convert satoshis to BTC: 1 BTC = 100,000,000 satoshis (1e8)
    return (funded - spent) / 1e8;
  } catch (err) {
    console.error(
      `[bitcoin] getBitcoinBalance failed for address ${safeAddress}:`,
      err
    );
    // Never throw to caller — return 0 so API route does not crash
    return 0;
  }
}
```
